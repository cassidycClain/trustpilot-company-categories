file_paths = []

def parse_directory_structure(lines, base_path="trustpilot-company-categories-scraper"):
 """
 Parse the directory structure string and return all file paths based on indentation.
 """
 stack = [(base_path, 0)] # (current path, indentation level)
 for line in lines:
 # Determine the current level by counting leading spaces
 indent_level = len(line) - len(line.lstrip(' '))
 current_path = line.strip()

 # Determine the full path based on indentation
 full_path = stack[indent_level][0] + "/" + current_path if indent_level > 0 else current_path

 # If the line represents a file (not a folder), add it to the list
 if "." in current_path:
 file_paths.append(full_path)

 # Update the stack for directory levels
 if current_path.endswith('/'):
 stack.append((full_path, indent_level + 1))
 else:
 stack[indent_level] = (stack[indent_level][0], indent_level)

# Run the parser on the structure
parse_directory_structure(directory_structure_lines)

# Displaying the output (final structure)
print(file_paths)
